const Http = new XMLHttpRequest();
let botToken = "你的_BOT_TOKEN"
let ChatID = "目标_CHAT_ID"
let txtmessage = "来自代理的 Hello"

var url = 'https://telegram-api-proxy-anonymous.pages.dev/api/bot' + botToken + 
          '/sendMessage?text=' + txtmessage + 
          '&chat_id=' + ChatID + 
          "&parse_mode=Markdown&disable_web_page_preview=True";

Http.open("GET", url);
Http.send();

Http.onreadystatechange = function() {
    if(this.readyState == 4 && this.status == 200) {
        console.log("消息发送成功");
    }
}
                

import requests

def send_telegram_msg(msg):
    token = "你的_BOT_TOKEN"
    chat_id = "目标_CHAT_ID"
    url = "https://telegram-api-proxy-anonymous.pages.dev/api/bot" + token + "/sendMessage"
    
    payload = {
        "text": msg,
        "chat_id": chat_id,
        "parse_mode": "Markdown",
        "disable_web_page_preview": True
    }
    
    response = requests.post(url, json=payload)
    return response.json()

user_msg = input("请输入要发送的消息：")
result = send_telegram_msg(user_msg)
print("消息发送成功！")
